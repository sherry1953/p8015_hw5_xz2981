add = function(x) {

fill_missing = function(x) {
  
  if (is.numeric(x)) {
    x = replace(x, which(is.na(x)), mean(x, na.rm = TRUE))
    }
  
  else {x = ifelse(is.na(x), "virginica", x)}

  return(x)

}
